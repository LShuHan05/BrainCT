16. AI影像分析结果表：ai_image_analysis
-- 存储 BrainCT 等 AI 模型的识别结果
CREATE TABLE ai_image_analysis (
    id                    INT PRIMARY KEY AUTO_INCREMENT,
    register_id           INT NOT NULL,              -- 关联挂号记录
    patient_id            VARCHAR(64) NOT NULL,       -- 病历号

    -- 影像信息
    image_type            VARCHAR(32) NOT NULL,       -- 影像类型: CT/MRI/X-ray
    image_path            VARCHAR(512),               -- 原始影像路径
    image_series_uid      VARCHAR(128),               -- DICOM SeriesInstanceUID

    -- AI模型信息
    model_name            VARCHAR(64) NOT NULL,       -- 模型名称: BrainCT-v1
    model_version         VARCHAR(32),                -- 模型版本
    analysis_type         VARCHAR(32) NOT NULL,       -- 分析类型: lesion/artifact/quality

    -- 识别结果（结构化数据）
    findings_json         JSON,                       -- AI识别的详细结果
    /*
    示例 JSON 结构:
    {
      "lesions": [
        {
          "type": "ICH",
          "probability": 0.95,
          "location": "right_hemisphere",
          "size_mm": 15.3,
          "coordinates": {"x": 120, "y": 85, "z": 45}
        }
      ],
      "emergency_level": "B2",
      "confidence_score": 0.92
    }
    */

    -- 紧急程度分级
    emergency_level       VARCHAR(16),                -- B1-轻症/B2-中重度
    urgency_score         DECIMAL(3,2),               -- 紧急度评分 0-1

    -- 置信度指标
    overall_confidence    DECIMAL(3,2),               -- 整体置信度 0-1
    precision_score       DECIMAL(3,2),               -- 精确率
    recall_score          DECIMAL(3,2),               -- 召回率

    -- 处理状态
    status                VARCHAR(32) DEFAULT 'pending', -- pending/completed/failed/reviewed
    processing_time_ms    INT,                        -- 处理耗时(毫秒)

    -- 医生审核
    reviewed_by           INT,                        -- 审核医生ID
    review_status         VARCHAR(32) DEFAULT 'unreviewed', -- unreviewed/confirmed/corrected
    review_comments       TEXT,                       -- 审核意见
    reviewed_at           DATETIME,                   -- 审核时间

    -- 时间戳
    created_at            DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at            DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (register_id) REFERENCES register(id),
    FOREIGN KEY (reviewed_by) REFERENCES employee(id),
    INDEX idx_patient (patient_id),
    INDEX idx_status (status),
    INDEX idx_emergency (emergency_level)
);


17. AI诊断建议表：ai_diagnosis_suggestion
-- 存储 AI 生成的诊断建议和处理意见
CREATE TABLE ai_diagnosis_suggestion (
    id                    INT PRIMARY KEY AUTO_INCREMENT,
    register_id           INT NOT NULL,               -- 关联挂号记录
    analysis_id           INT,                        -- 关联影像分析结果

    -- AI 诊断内容
    primary_diagnosis     VARCHAR(512),               -- 主要诊断
    differential_diagnosis JSON,                      -- 鉴别诊断列表
    /*
    示例:
    [
      {"disease": "脑实质出血", "probability": 0.85},
      {"disease": "脑梗死", "probability": 0.12}
    ]
    */

    suggested_tests       JSON,                       -- 建议的检查项目
    suggested_treatments  JSON,                       -- 建议的治疗方案
    medication_suggestions JSON,                      -- 用药建议

    -- 风险预警
    risk_factors          JSON,                       -- 风险因素
    contraindications     TEXT,                       -- 禁忌症提醒

    -- 生成信息
    llm_model             VARCHAR(64),                -- 使用的大模型: deepseek/gpt
    generation_prompt     TEXT,                       -- 使用的提示词
    raw_llm_response      LONGTEXT,                   -- 大模型原始响应

    -- 医生采纳情况
    adoption_status       VARCHAR(32) DEFAULT 'pending', -- pending/adopted/partially_adopted/rejected
    adopted_by            INT,                        -- 采纳医生ID
    adoption_notes        TEXT,                       -- 采纳说明
    adopted_at            DATETIME,                   -- 采纳时间

    created_at            DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (register_id) REFERENCES register(id),
    FOREIGN KEY (analysis_id) REFERENCES ai_image_analysis(id),
    FOREIGN KEY (adopted_by) REFERENCES employee(id)
);


18. AI医患对话记录表：ai_conversation
-- 存储患者与 AI 助手的对话历史
CREATE TABLE ai_conversation (
    id                    BIGINT PRIMARY KEY AUTO_INCREMENT,
    conversation_id       VARCHAR(64) NOT NULL,       -- 会话ID（UUID）

    -- 参与者信息
    participant_type      VARCHAR(16) NOT NULL,       -- patient/doctor/admin
    participant_id        VARCHAR(64) NOT NULL,       -- 用户ID（病历号或医生ID）
    register_id           INT,                        -- 关联挂号记录（可选）

    -- 对话内容
    role                  VARCHAR(16) NOT NULL,       -- user/assistant/system
    message_type          VARCHAR(32) DEFAULT 'text', -- text/image/result_card

    content               LONGTEXT NOT NULL,          -- 消息内容
    content_summary       VARCHAR(512),               -- 内容摘要（用于快速展示）

    -- 多媒体附件
    attachments_json      JSON,                       -- 附件信息
    /*
    示例:
    {
      "images": ["path/to/image1.jpg"],
      "reports": ["path/to/report.pdf"]
    }
    */

    -- AI 元数据
    llm_model             VARCHAR(64),                -- 使用的模型
    tokens_used           INT,                        -- 消耗的 token 数
    response_time_ms      INT,                        -- 响应时间
    temperature           DECIMAL(3,2),               -- 生成温度参数

    -- 上下文信息
    context_snapshot      JSON,                       -- 对话上下文快照
    intent_classification VARCHAR(64),                -- 意图分类: inquiry/symptom_report/follow_up

    -- 质量评估
    quality_score         DECIMAL(3,2),               -- 回答质量评分
    user_feedback         VARCHAR(32),                -- 用户反馈: helpful/not_helpful
    flagged               BOOLEAN DEFAULT FALSE,      -- 是否标记为异常

    -- 时间戳
    created_at            DATETIME DEFAULT CURRENT_TIMESTAMP,

    INDEX idx_conversation (conversation_id),
    INDEX idx_participant (participant_type, participant_id),
    INDEX idx_register (register_id),
    INDEX idx_created (created_at)
);

    evidence_level        VARCHAR(16),                -- 证据等级: A/B/C
    publication_date      DATE,                       -- 发布日期

    -- 适用性
    applicable_departments JSON,                      -- 适用科室
    tags                  JSON,                       -- 标签

    -- 使用统计
    view_count            INT DEFAULT 0,              -- 查看次数
    citation_count        INT DEFAULT 0,              -- 引用次数

    updated_at            DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (verified_by) REFERENCES employee(id),
    INDEX idx_category (category),
    INDEX idx_tags ((CAST(tags AS CHAR(512)))),
    FULLTEXT INDEX ft_content (title, content)

23. AI 性能评估表：ai_performance_metrics

CREATE TABLE ai_performance_metrics (

    -- 评估对象
    model_name            VARCHAR(64) NOT NULL,
    model_version         VARCHAR(32) NOT NULL,
    evaluation_type       VARCHAR(32),                -- accuracy/diagnostic_quality/user_satisfaction

    confusion_matrix      JSON,                       -- 混淆矩阵

    -- 评估信息
    evaluated_by          INT,                        -- 评估者ID
    evaluation_method     VARCHAR(64),                -- 评估方法
    evaluation_date       DATE NOT NULL,              -- 评估日期

    notes                 TEXT,                       -- 备注

    created_at            DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (evaluated_by) REFERENCES employee(id),

    INDEX idx_date (evaluation_date)
